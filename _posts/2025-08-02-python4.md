```python3
#!/usr/bin/env python3

nombre = input("\nTemperatura en Celsius: ")

try:
    temperature_c = float(nombre)
    temperature_f= (temperature_c * 9/5) + 32
    print(f"\nTemperatura en Fahrenheit: {temperature_f:.2f}°F")
except ValueError:
    print("\nError: Por favor, ingrese un número válido.")
```
